82  //-----------------------------------------------------------------------------
83  //! Waits for all the other threads to reach the barrier.
84  //-----------------------------------------------------------------------------
85  ALPAKA_FN_ACC_NO_CUDA auto wait()
86  -> void
87  {
88  TSize const generationWhenEnteredTheWait = m_generation;
89 #ifdef ALPAKA_THREAD_BARRIER_DISABLE_SPINLOCK
90  std::unique_lock<std::mutex> lock(m_mtxBarrier);
91 #endif
92  if(--m_curThreadCount == 0)
93  {
94  m_curThreadCount = m_threadCount;
95  ++m_generation;
96 #ifdef ALPAKA_THREAD_BARRIER_DISABLE_SPINLOCK
97  m_cvAllThreadsReachedBarrier.notify_all();
98 #endif
99  }
100  else
101  {
102 #ifdef ALPAKA_THREAD_BARRIER_DISABLE_SPINLOCK
103  m_cvAllThreadsReachedBarrier.wait(lock, [this, generationWhenEnteredTheWait] { return generationWhenEnteredTheWait != m_generation; });
104 #else
105  while(generationWhenEnteredTheWait == m_generation)
106  {
107  std::this_thread::yield();
108  }
109 #endif
110  }
111  }
112 
113  private:
114 #ifdef ALPAKA_THREAD_BARRIER_DISABLE_SPINLOCK
115  std::mutex m_mtxBarrier;
116  std::condition_variable m_cvAllThreadsReachedBarrier;
117 #endif
118  const TSize m_threadCount;
119 #ifdef ALPAKA_THREAD_BARRIER_DISABLE_SPINLOCK
120  TSize m_curThreadCount;
121  TSize m_generation;
122 #else
123  std::atomic<TSize> m_curThreadCount;
