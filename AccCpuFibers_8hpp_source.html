66  //! The CPU fibers accelerator.
67  //!
68  //! This accelerator allows parallel kernel execution on a CPU device.
69  //! It uses boost::fibers to implement the cooperative parallelism.
70  //! By using fibers the shared memory can reside in the closest memory/cache available.
71  //! Furthermore there is no false sharing between neighboring threads as it is the case in real multi-threading.
72  //#############################################################################
73  template<
74  typename TDim,
75  typename TSize>
76  class AccCpuFibers final :
77  public workdiv::WorkDivMembers<TDim, TSize>,
78  public idx::gb::IdxGbRef<TDim, TSize>,
79  public idx::bt::IdxBtRefFiberIdMap<TDim, TSize>,
80  public atomic::AtomicNoOp,
81  public math::MathStl,
82  public block::shared::BlockSharedAllocMasterSync,
83  public block::sync::BlockSyncFiberIdMapBarrier<TSize>,
84  public rand::RandStl
85  {
86  public:
87  // Partial specialization with the correct TDim and TSize is not allowed.
88  template<
89  typename TDim2,
90  typename TSize2,
91  typename TKernelFnObj,
92  typename... TArgs>
93  friend class ::alpaka::exec::ExecCpuFibers;
94 
95  private:
96  //-----------------------------------------------------------------------------
97  //! Constructor.
98  //-----------------------------------------------------------------------------
99  template<
100  typename TWorkDiv>
101  ALPAKA_FN_ACC_NO_CUDA AccCpuFibers(
102  TWorkDiv const & workDiv) :
103  workdiv::WorkDivMembers<TDim, TSize>(workDiv),
104  idx::gb::IdxGbRef<TDim, TSize>(m_gridBlockIdx),
105  idx::bt::IdxBtRefFiberIdMap<TDim, TSize>(m_fibersToIndices),
106  atomic::AtomicNoOp(),
107  math::MathStl(),
108  block::shared::BlockSharedAllocMasterSync(
109  [this](){block::sync::syncBlockThreads(*this);},
110  [this](){return (m_masterFiberId == boost::this_fiber::get_id());}),
111  block::sync::BlockSyncFiberIdMapBarrier<TSize>(
112  m_blockThreadCount,
113  m_fibersToBarrier),
114  rand::RandStl(),
115  m_gridBlockIdx(Vec<TDim, TSize>::zeros()),
116  m_blockThreadCount(workdiv::getWorkDiv<Block, Threads>(workDiv).prod())
117  {}
118 
119  public:
120  //-----------------------------------------------------------------------------
121  //! Copy constructor.
122  //-----------------------------------------------------------------------------
123  ALPAKA_FN_ACC_NO_CUDA AccCpuFibers(AccCpuFibers const &) = delete;

141  //-----------------------------------------------------------------------------
142  //! \return The pointer to the externally allocated block shared memory.
143  //-----------------------------------------------------------------------------
144  template<
145  typename T>
146  ALPAKA_FN_ACC_NO_CUDA auto getBlockSharedExternMem() const
147  -> T *
148  {
149  return reinterpret_cast<T*>(m_externalSharedMem.get());
150  }
151 
152  private:
153  // getIdx
154  typename idx::bt::IdxBtRefFiberIdMap<TDim, TSize>::FiberIdToIdxMap mutable m_fibersToIndices; //!< The mapping of fibers id's to indices.
155  alignas(16u) Vec<TDim, TSize> mutable m_gridBlockIdx; //!< The index of the currently executed block.
156 
157  // syncBlockThreads
158  TSize const m_blockThreadCount; //!< The number of threads per block the barrier has to wait for.
159  std::map<
160  boost::fibers::fiber::id,
161  TSize> mutable m_fibersToBarrier; //!< The mapping of fibers id's to their current barrier.
162  //!< We have to keep the current and the last barrier because one of the fibers can reach the next barrier before another fiber was wakeup from the last one and has checked if it can run.
163 
164  // allocBlockSharedArr
165  boost::fibers::fiber::id mutable m_masterFiberId; //!< The id of the master fiber.
166 
167  // getBlockSharedExternMem
168  std::unique_ptr<uint8_t, boost::alignment::aligned_delete> mutable m_externalSharedMem; //!< External block shared memory.
169  };
