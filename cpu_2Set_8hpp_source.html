63  typename TBuf,
64  typename TExtents>
65  ALPAKA_FN_HOST static auto set(
66  TBuf & buf,
67  std::uint8_t const & byte,
68  TExtents const & extents)
69  -> void
70  {
71  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
72 
73  using Elem = mem::view::ElemT<TBuf>;
74 
75  static_assert(
76  dim::Dim<TBuf>::value == dim::Dim<TExtents>::value,
77  "The destination buffer and the extents are required to have the same dimensionality!");
78 
79  auto const uiExtentWidth(extent::getWidth(extents));
80  auto const uiExtentHeight(extent::getHeight(extents));
81  auto const uiExtentDepth(extent::getDepth(extents));
82  auto const uiDstWidth(extent::getWidth(buf));
83  auto const uiDstHeight(extent::getHeight(buf));
84 #if (!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
85  auto const uiDstDepth(extent::getDepth(buf));
86 #endif
87  assert(uiExtentWidth <= uiDstWidth);
88  assert(uiExtentHeight <= uiDstHeight);
89  assert(uiExtentDepth <= uiDstDepth);
90 
91  auto const uiExtentWidthBytes(uiExtentWidth * sizeof(Elem));
92  auto const uiDstPitchBytes(mem::view::getPitchBytes<dim::Dim<TBuf>::value - 1u>(buf));
93  assert(uiExtentWidthBytes <= uiDstPitchBytes);
94 
95  auto const pDstNative(reinterpret_cast<std::uint8_t *>(mem::view::getPtrNative(buf)));
96  auto const uiDstSliceSizeBytes(uiDstPitchBytes * uiDstHeight);
97 
98  auto const & dstBuf(mem::view::getBuf(buf));
99  auto const uiDstBufWidth(extent::getWidth(dstBuf));
100  auto const uiDstBufHeight(extent::getHeight(dstBuf));
101 
102  int iByte(static_cast<int>(byte));
103 
104 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
105  std::cout << BOOST_CURRENT_FUNCTION
106  << " ew: " << uiExtentWidth
107  << " eh: " << uiExtentHeight
108  << " ed: " << uiExtentDepth
109  << " ewb: " << uiExtentWidthBytes
110  << " dw: " << uiDstWidth
111  << " dh: " << uiDstHeight
112  << " dd: " << uiDstDepth
113  << " dptr: " << reinterpret_cast<void *>(pDstNative)
114  << " dpitchb: " << uiDstPitchBytes
115  << " dbasew: " << uiDstBufWidth
116  << " dbaseh: " << uiDstBufHeight
117  << std::endl;
118 #endif
119  // If:
120  // - the set extents width and height are identical to the dst extents width and height
121  // -> we can set the whole memory at once overwriting the pitch bytes
122  if((uiExtentWidth == uiDstWidth)
123  && (uiExtentHeight == uiDstHeight)
124  && (uiExtentWidth == uiDstBufWidth)
125  && (uiExtentHeight == uiDstBufHeight))
126  {
127  std::memset(
128  reinterpret_cast<void *>(pDstNative),
129  iByte,
130  uiDstSliceSizeBytes*uiExtentDepth);
131  }
132  else
133  {
134  for(auto z(decltype(uiExtentDepth)(0)); z < uiExtentDepth; ++z)
135  {
136  // If:
137  // - the set extents width is identical to the dst extents width
138  // -> we can set whole slices at once overwriting the pitch bytes
139  if((uiExtentWidth == uiDstWidth)
140  && (uiExtentWidth == uiDstBufWidth))
141  {
142  std::memset(
143  reinterpret_cast<void *>(pDstNative + z*uiDstSliceSizeBytes),
144  iByte,
145  uiDstPitchBytes*uiExtentHeight);
146  }
147  else
148  {
149  for(auto y(decltype(uiExtentHeight)(0)); y < uiExtentHeight; ++y)
150  {
151  std::memset(
152  reinterpret_cast<void *>(pDstNative + y*uiDstPitchBytes + z*uiDstSliceSizeBytes),
153  iByte,
154  uiExtentWidthBytes);
155  }
156  }
157  }
158  }
159  }
