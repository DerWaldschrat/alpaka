129  //! Executes the kernel function object.
130  //-----------------------------------------------------------------------------
131  template<
132  typename TKernelFnObj,
133  typename... TArgs>
134  ALPAKA_FN_HOST auto operator()(
135  // \NOTE: No const reference (const &) is allowed as the parameter type because the kernel launch language extension expects the arguments by value.
136  // This forces the type of a float argument given with std::forward to this function to be of type float instead of e.g. "float const & __ptr64" (MSVC).
137  // If not given by value, the kernel launch code does not copy the value but the pointer to the value location.
138  TKernelFnObj kernelFnObj,
139  TArgs ... args) const
140  -> void
141  {
142  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
143 
144 #if (!__GLIBCXX__) // libstdc++ even for gcc-4.9 does not support std::is_trivially_copyable.
145  static_assert(
146  boost::mpl::and_<
147  // This true_ is required for the zero argument case because and_ requires at least two arguments.
148  boost::mpl::true_,
149  std::is_trivially_copyable<
150  TKernelFnObj>,
151  boost::mpl::apply<
152  std::is_trivially_copyable<
153  boost::mpl::_1>,
154  TArgs>...
155  >::value,
156  "The given kernel function object and its arguments have to fulfill is_trivially_copyable!");
157 #endif
158  // TODO: Check that (sizeof(TKernelFnObj) * m_3uiBlockThreadExtents.prod()) < available memory size
159 
160 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
161  //std::size_t uiPrintfFifoSize;
162  //cudaDeviceGetLimit(&uiPrintfFifoSize, cudaLimitPrintfFifoSize);
163  //std::cout << BOOST_CURRENT_FUNCTION << "INFO: uiPrintfFifoSize: " << uiPrintfFifoSize << std::endl;
164  //cudaDeviceSetLimit(cudaLimitPrintfFifoSize, uiPrintfFifoSize*10);
165  //cudaDeviceGetLimit(&uiPrintfFifoSize, cudaLimitPrintfFifoSize);
166  //std::cout << BOOST_CURRENT_FUNCTION << "INFO: uiPrintfFifoSize: " << uiPrintfFifoSize << std::endl;
167 #endif
168 
169  auto const vuiGridBlockExtents(
170  workdiv::getWorkDiv<Grid, Blocks>(
171  *static_cast<workdiv::WorkDivMembers<TDim, TSize> const *>(this)));
172  auto const vuiBlockThreadExtents(
173  workdiv::getWorkDiv<Block, Threads>(
174  *static_cast<workdiv::WorkDivMembers<TDim, TSize> const *>(this)));
175 
176  dim3 gridDim(1u, 1u, 1u);
177  dim3 blockDim(1u, 1u, 1u);
178  // \FIXME: CUDA currently supports a maximum of 3 dimensions!
179  for(auto i(static_cast<typename TDim::value_type>(0)); i<std::min(static_cast<typename TDim::value_type>(3), TDim::value); ++i)
180  {
181  reinterpret_cast<unsigned int *>(&gridDim)[i] = vuiGridBlockExtents[TDim::value-1u-i];
182  reinterpret_cast<unsigned int *>(&blockDim)[i] = vuiBlockThreadExtents[TDim::value-1u-i];
183  }
184  // Assert that all extents of the higher dimensions are 1!
185  for(auto i(std::min(static_cast<typename TDim::value_type>(3), TDim::value)); i<TDim::value; ++i)
186  {
187  assert(vuiGridBlockExtents[TDim::value-1u-i] == 1);
188  assert(vuiBlockThreadExtents[TDim::value-1u-i] == 1);
189  }
190 
191 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
192  std::cout << BOOST_CURRENT_FUNCTION << "gridDim: " << gridDim.z << " " << gridDim.y << " " << gridDim.x << std::endl;
193  std::cout << BOOST_CURRENT_FUNCTION << "blockDim: " << blockDim.z << " " << blockDim.y << " " << blockDim.x << std::endl;
194 #endif
195 
196  // Get the size of the block shared extern memory.
197  auto const uiBlockSharedExternMemSizeBytes(
198  kernel::getBlockSharedExternMemSizeBytes<
199  typename std::decay<TKernelFnObj>::type,
200  acc::AccGpuCudaRt<TDim, TSize>>(
201  vuiBlockThreadExtents,
202  args...));
203 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
204  // Log the block shared memory size.
205  std::cout << BOOST_CURRENT_FUNCTION
206  << " BlockSharedExternMemSizeBytes: " << uiBlockSharedExternMemSizeBytes << " B"
207  << std::endl;
208 #endif
209 
210 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
211  // Log the function attributes.
212  cudaFuncAttributes funcAttrs;
213  cudaFuncGetAttributes(&funcAttrs, cuda::detail::cudaKernel<TDim, TSize, TKernelFnObj, TArgs...>);
214  std::cout << BOOST_CURRENT_FUNCTION
215  << " binaryVersion: " << funcAttrs.binaryVersion
216  << " constSizeBytes: " << funcAttrs.constSizeBytes << " B"
217  << " localSizeBytes: " << funcAttrs.localSizeBytes << " B"
218  << " maxThreadsPerBlock: " << funcAttrs.maxThreadsPerBlock
219  << " numRegs: " << funcAttrs.numRegs
220  << " ptxVersion: " << funcAttrs.ptxVersion
221  << " sharedSizeBytes: " << funcAttrs.sharedSizeBytes << " B"
222  << std::endl;
223 #endif
224 
225  // Set the current device.
226  ALPAKA_CUDA_RT_CHECK(cudaSetDevice(
227  m_Stream.m_spStreamCudaImpl->m_Dev.m_iDevice));
228  // Enqueue the kernel execution.
229  cuda::detail::cudaKernel<TDim, TSize, TKernelFnObj, TArgs...><<<
230  gridDim,
231  blockDim,
232  uiBlockSharedExternMemSizeBytes,
233  m_Stream.m_spStreamCudaImpl->m_CudaStream>>>(
234  kernelFnObj,
235  args...);
236 
237 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
238  // Wait for the kernel execution to finish but do not check error return of this call.
239  // Do not use the alpaka::wait method because it checks the error itself but we want to give a custom error message.
240  cudaStreamSynchronize(m_Stream.m_spStreamCudaImpl->m_CudaStream);
241  //cudaDeviceSynchronize();
242  cudaError_t const error(cudaGetLastError());
243  if(error != cudaSuccess)
244  {
245  std::string const sError("The execution of kernel '" + std::string(typeid(TKernelFnObj).name()) + "' failed with error: '" + std::string(cudaGetErrorString(error)) + "'");
246  std::cerr << sError << std::endl;
247  ALPAKA_DEBUG_BREAK;
248  throw std::runtime_error(sError);
249  }
250 #endif
