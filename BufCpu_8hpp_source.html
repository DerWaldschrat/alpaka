514  {
515  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
516 
517  if(!mem::buf::isPinned(buf))
518  {
519 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA
520  if(buf.m_spBufCpuImpl->m_extentElements.prod() != 0)
521  {
522  // - cudaHostRegisterDefault:
523  // See http://cgi.cs.indiana.edu/~nhusted/dokuwiki/doku.php?id=programming:cudaperformance1
524  // - cudaHostRegisterPortable:
525  // The memory returned by this call will be considered as pinned memory by all CUDA contexts, not just the one that performed the allocation.
526  ALPAKA_CUDA_RT_CHECK_IGNORE(
527  cudaHostRegister(
528  const_cast<void *>(reinterpret_cast<void const *>(mem::view::getPtrNative(buf))),
529  extent::getProductOfExtent(buf) * sizeof(elem::Elem<buf::BufCpu<TElem, TDim, TSize>>),
530  cudaHostRegisterDefault),
531  cudaErrorHostMemoryAlreadyRegistered);
532 
533  buf.m_spBufCpuImpl->m_bPinned = true;
534  }
535 #else
536  static_assert(
537  meta::DependentFalseType<TElem>::value,
538  "Memory pinning of BufCpu is not implemented when CUDA is not enabled!");
539 #endif
540  }
541  }
