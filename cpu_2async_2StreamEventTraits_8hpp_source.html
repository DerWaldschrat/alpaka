55  //! The CPU device stream enqueue trait specialization.
56  //#############################################################################
57  template<>
58  struct StreamEnqueue<
59  std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl>,
60  event::EventCpuAsync>
61  {
62  ALPAKA_FN_HOST static auto streamEnqueue(
63  std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl> & spStreamImpl,
64  event::EventCpuAsync & event)
65  -> void
66  {
67  // Copy the shared pointer of the event implementation.
68  // This is forwarded to the lambda that is enqueued into the stream to ensure that the event implementation is alive as long as it is enqueued.
69  auto spEventCpuImpl(event.m_spEventCpuAsyncImpl);
70 
71  // Setting the event state and enqueuing it has to be atomic.
72  std::lock_guard<std::mutex> lk(spEventCpuImpl->m_Mutex);
73 
74  // This is a invariant: If the event is ready (not enqueued) there can not be anybody waiting for it.
75  assert(!(spEventCpuImpl->m_bIsReady && spEventCpuImpl->m_bIsWaitedFor));
76 
77  // If it is enqueued ...
78  if(!spEventCpuImpl->m_bIsReady)
79  {
80  // ... and somebody is waiting for it, it can NOT be re-enqueued.
81  if(spEventCpuImpl->m_bIsWaitedFor)
82  {
83 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
84  std::cout << BOOST_CURRENT_FUNCTION << "WARNING: The event to enqueue is already enqueued AND waited on. It can NOT be re-enqueued!" << std::endl;
85 #endif
86  return;
87  }
88  // ... and was enqueued before, increment the cancel counter.
89  else
90  {
91  ++spEventCpuImpl->m_uiNumCanceledEnqueues;
92  }
93  }
94  // If it is not enqueued, set its state to enqueued.
95  else
96  {
97  spEventCpuImpl->m_bIsReady = false;
98  }
99 
100  // Enqueue a task that only resets the events flag if it is completed.
101  spStreamImpl->m_workerThread.enqueueTask(
102  [spEventCpuImpl]()
103  {
104  {
105  std::lock_guard<std::mutex> lk(spEventCpuImpl->m_Mutex);
106  // Nothing to do if it has been re-enqueued to a later position in the queue.
107  if(spEventCpuImpl->m_uiNumCanceledEnqueues > 0)
108  {
109  --spEventCpuImpl->m_uiNumCanceledEnqueues;
110  return;
111  }
112  else
113  {
114  spEventCpuImpl->m_bIsWaitedFor = false;
115  spEventCpuImpl->m_bIsReady = true;
116  }
117  }
118  spEventCpuImpl->m_ConditionVariable.notify_all();
119  });
120  }
121  };
