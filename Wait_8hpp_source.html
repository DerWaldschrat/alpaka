38  //! Blocks until the device has completed all preceding requested tasks.
39  //! Tasks that are enqueued or streams that are created after this call is made are not waited for.
40  //#############################################################################

45  //-----------------------------------------------------------------------------

48  ALPAKA_FN_HOST static auto currentThreadWaitFor(
49  dev::DevCpu const & dev)
50  -> void
51  {
52  ALPAKA_DEBUG_FULL_LOG_SCOPE;
53 
54  // Get all the streams on the device at the time of invocation.
55  // All streams added afterwards are ignored.
56  auto vspStreams(
57  dev.m_spDevCpuImpl->GetAllAsyncStreamImpls());
58 
59  // Enqueue an event in every asynchronous stream on the device.
60  // \FIXME: This should be done atomically for all streams.
61  // Furthermore there should not even be a chance to enqueue something between getting the streams and adding our wait events!
62  std::vector<event::EventCpu> vEvents;
63  for(auto && spStream : vspStreams)
64  {
65  vEvents.emplace_back(dev);
66  stream::enqueue(spStream, vEvents.back());
67  }
68 
69  // Now wait for all the events.
70  for(auto && event : vEvents)
71  {
72  wait::wait(event);
73  }
74  }
75  };
