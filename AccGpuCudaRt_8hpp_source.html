196  std::numeric_limits<TSize>::max(),
197  // m_blockThreadExtentMax

200  static_cast<TSize>(cudaDevProp.maxThreadsDim[2]),
201  static_cast<TSize>(cudaDevProp.maxThreadsDim[1]),
202  static_cast<TSize>(cudaDevProp.maxThreadsDim[0]))),
203  // m_blockThreadCountMax
