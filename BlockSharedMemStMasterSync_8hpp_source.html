109  ALPAKA_FN_ACC_NO_CUDA static auto allocVar(
110  block::shared::st::BlockSharedMemStMasterSync const & blockSharedMemSt)
111  -> T &
112  {
113  // Assure that all threads have executed the return of the last allocBlockSharedArr function (if there was one before).
114  blockSharedMemSt.m_syncFn();
115 
116  if(blockSharedMemSt.m_isMasterThreadFn())
117  {
118  blockSharedMemSt.m_sharedAllocs.emplace_back(
119  reinterpret_cast<uint8_t *>(
120  boost::alignment::aligned_alloc(16u, sizeof(T))));
121  }
122  blockSharedMemSt.m_syncFn();
123 
124  return
125  std::ref(
126  *reinterpret_cast<T*>(
127  blockSharedMemSt.m_sharedAllocs.back().get()));
128  }
129  };
