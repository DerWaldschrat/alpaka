397  //#############################################################################
398  //! The general vectorization implementation.
399  //#############################################################################
400  template<
401  std::size_t TvectorizationCount>
402  struct Vectorize
403  {
404  //-----------------------------------------------------------------------------
405  //!
406  //-----------------------------------------------------------------------------
407  template<
408  typename TSize,
409  typename TFnObj>
410  ALPAKA_FN_HOST static auto vectorize(
411  TSize const & count,
412  TFnObj const & fnObj)
413  -> void
414  {
415 #ifdef __CUDACC__ // nvcc seems not to support constexpr correctly
416  static_assert(
417  (static_cast<TSize>(TvectorizationCount) > 0),
418  "The number of loop invocations to vectorize (TvectorizationCount) has to be greater zero!");
419 
420  TSize i(0);
421  while((count-i) >= static_cast<TSize>(TvectorizationCount))
422  {
423  for(TSize j(0); j<static_cast<TSize>(TvectorizationCount); ++j)
424  {
425  fnObj(i, j);
426  }
427  i += static_cast<TSize>(TvectorizationCount);
428  }
429 #else
430  // Cast it to the user defined type.
431  constexpr TSize vecRegElems(
432  static_cast<TSize>(TvectorizationCount));
433 
434  static_assert(
435  (vecRegElems > 0),
436  "The number of loop invocations to vectorize (TvectorizationCount) has to be greater zero!");
437 
438  // While we have more then TvectorizationCount elements left ...
439  TSize i(0);
440  while((count-i) >= vecRegElems)
441  {
442  // ... execute exactly TvectorizationCount invocations of the loop.
443  // This enables the compiler to optimize much better because of the compile-time loop bounds.
444  for(TSize j(0); j<vecRegElems; ++j)
445  {
446  fnObj(i, j);
447  }
448  i += vecRegElems;
449  }
450 #endif
451  // Execute the last (count % TvectorizationCount) invocations.
452  for(TSize j(0); j<(count-i); ++j)
453  {
454  fnObj(i, j);
455  }
456  }
457  };
