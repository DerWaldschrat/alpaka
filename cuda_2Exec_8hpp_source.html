125  //-----------------------------------------------------------------------------
126  //! Executes the kernel function object.
127  //-----------------------------------------------------------------------------
128  template<
129  typename TKernelFnObj,
130  typename... TArgs>
131  ALPAKA_FN_HOST auto operator()(
132  // \NOTE: No const reference (const &) is allowed as the parameter type because the kernel launch language extension expects the arguments by value.
133  // This forces the type of a float argument given with std::forward to this function to be of type float instead of e.g. "float const & __ptr64" (MSVC).
134  // If not given by value, the kernel launch code does not copy the value but the pointer to the value location.
135  TKernelFnObj kernelFnObj,
136  TArgs ... args) const
137  -> void
138  {
139  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
140 
141 #if (!__GLIBCXX__) // libstdc++ even for gcc-4.9 does not support std::is_trivially_copyable.
142  static_assert(
143  std::is_trivially_copyable<TKernelFnObj>::value,
144  "The given kernel function object has to fulfill is_trivially_copyable!");
145 #endif
146  // TODO: Check that (sizeof(TKernelFnObj) * m_3uiBlockThreadExtents.prod()) < available memory size
147 
148 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
149  //std::size_t uiPrintfFifoSize;
150  //cudaDeviceGetLimit(&uiPrintfFifoSize, cudaLimitPrintfFifoSize);
151  //std::cout << BOOST_CURRENT_FUNCTION << "INFO: uiPrintfFifoSize: " << uiPrintfFifoSize << std::endl;
152  //cudaDeviceSetLimit(cudaLimitPrintfFifoSize, uiPrintfFifoSize*10);
153  //cudaDeviceGetLimit(&uiPrintfFifoSize, cudaLimitPrintfFifoSize);
154  //std::cout << BOOST_CURRENT_FUNCTION << "INFO: uiPrintfFifoSize: " << uiPrintfFifoSize << std::endl;
155 #endif
156 
157  auto const vuiGridBlockExtents(
158  workdiv::getWorkDiv<Grid, Blocks>(
159  *static_cast<workdiv::WorkDivMembers<TDim, TSize> const *>(this)));
160  auto const vuiBlockThreadExtents(
161  workdiv::getWorkDiv<Block, Threads>(
162  *static_cast<workdiv::WorkDivMembers<TDim, TSize> const *>(this)));
163 
164  dim3 gridDim(1u, 1u, 1u);
165  dim3 blockDim(1u, 1u, 1u);
166  // \FIXME: CUDA currently supports a maximum of 3 dimensions!
167  for(auto i(static_cast<typename TDim::value_type>(0)); i<std::min(static_cast<typename TDim::value_type>(3), TDim::value); ++i)
168  {
169  reinterpret_cast<unsigned int *>(&gridDim)[i] = vuiGridBlockExtents[TDim::value-1u-i];
170  reinterpret_cast<unsigned int *>(&blockDim)[i] = vuiBlockThreadExtents[TDim::value-1u-i];
171  }
172  // Assert that all extents of the higher dimensions are 1!
173  for(auto i(std::min(static_cast<typename TDim::value_type>(3), TDim::value)); i<TDim::value; ++i)
174  {
175  assert(vuiGridBlockExtents[TDim::value-1u-i] == 1);
176  assert(vuiBlockThreadExtents[TDim::value-1u-i] == 1);
177  }
178 
179 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
180  std::cout << BOOST_CURRENT_FUNCTION << "gridDim: " << gridDim.z << " " << gridDim.y << " " << gridDim.x << std::endl;
181  std::cout << BOOST_CURRENT_FUNCTION << "blockDim: " << blockDim.z << " " << blockDim.y << " " << blockDim.x << std::endl;
182 #endif
183 
184  // Get the size of the block shared extern memory.
185  auto const uiBlockSharedExternMemSizeBytes(
186  kernel::getBlockSharedExternMemSizeBytes<
187  typename std::decay<TKernelFnObj>::type,
188  AccGpuCuda<TDim, TSize>>(
189  vuiBlockThreadExtents,
190  args...));
191 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
192  // Log the block shared memory size.
193  std::cout << BOOST_CURRENT_FUNCTION
194  << " BlockSharedExternMemSizeBytes: " << uiBlockSharedExternMemSizeBytes << " B"
195  << std::endl;
196 #endif
197 
198 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
199  // Log the function attributes.
200  cudaFuncAttributes funcAttrs;
201  cudaFuncGetAttributes(&funcAttrs, cuda::detail::cudaKernel<TDim, TSize, TKernelFnObj, TArgs...>);
202  std::cout << BOOST_CURRENT_FUNCTION
203  << " binaryVersion: " << funcAttrs.binaryVersion
204  << " constSizeBytes: " << funcAttrs.constSizeBytes << " B"
205  << " localSizeBytes: " << funcAttrs.localSizeBytes << " B"
206  << " maxThreadsPerBlock: " << funcAttrs.maxThreadsPerBlock
207  << " numRegs: " << funcAttrs.numRegs
208  << " ptxVersion: " << funcAttrs.ptxVersion
209  << " sharedSizeBytes: " << funcAttrs.sharedSizeBytes << " B"
210  << std::endl;
211 #endif
212 
213  // Set the current device.
214  ALPAKA_CUDA_RT_CHECK(cudaSetDevice(
215  m_Stream.m_spStreamCudaImpl->m_Dev.m_iDevice));
216  // Enqueue the kernel execution.
217  cuda::detail::cudaKernel<TDim, TSize, TKernelFnObj, TArgs...><<<
218  gridDim,
219  blockDim,
220  uiBlockSharedExternMemSizeBytes,
221  m_Stream.m_spStreamCudaImpl->m_CudaStream>>>(
222  kernelFnObj,
223  args...);
224 
225 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
226  // Wait for the kernel execution to finish but do not check error return of this call.
227  // Do not use the alpaka::wait method because it checks the error itself but we want to give a custom error message.
228  cudaStreamSynchronize(m_Stream.m_spStreamCudaImpl->m_CudaStream);
229  //cudaDeviceSynchronize();
230  cudaError_t const error(cudaGetLastError());
231  if(error != cudaSuccess)
232  {
233  std::string const sError("The execution of kernel '" + std::string(typeid(TKernelFnObj).name()) + "' failed with error: '" + std::string(cudaGetErrorString(error)) + "'");
234  std::cerr << sError << std::endl;
235  ALPAKA_DEBUG_BREAK;
236  throw std::runtime_error(sError);
237  }
238 #endif
239  }
