52  //! The CPU device memory copy task base.
53  //!
54  //! Copies from CPU memory into CPU memory.
55  //#############################################################################
56  template<
57  typename TDim,
58  typename TViewDst,
59  typename TViewSrc,
60  typename TExtent>
61  struct TaskCopyBase
62  {
63  using ExtentSize = size::Size<TExtent>;
64  using DstSize = size::Size<TViewDst>;
65  using SrcSize = size::Size<TViewSrc>;
66  using Elem = elem::Elem<TViewSrc>;
67 
68  static_assert(
69  dim::Dim<TViewDst>::value == dim::Dim<TViewSrc>::value,
70  "The source and the destination view are required to have the same dimensionality!");
71  static_assert(
72  dim::Dim<TViewDst>::value == dim::Dim<TExtent>::value,
73  "The views and the extent are required to have the same dimensionality!");
74  static_assert(
75  dim::Dim<TViewDst>::value == TDim::value,
76  "The destination view and the input TDim are required to have the same dimensionality!");
77 
78  static_assert(
79  meta::IsIntegralSuperset<DstSize, ExtentSize>::value,
80  "The destination view and the extent are required to have compatible size type!");
81  static_assert(
82  meta::IsIntegralSuperset<SrcSize, ExtentSize>::value,
83  "The source view and the extent are required to have compatible size type!");
84 
85  static_assert(
86  std::is_same<elem::Elem<TViewDst>, typename std::remove_const<elem::Elem<TViewSrc>>::type>::value,
87  "The source and the destination view are required to have the same element type!");
88 
89  //-----------------------------------------------------------------------------
90  //! Constructor.
91  //-----------------------------------------------------------------------------
92  TaskCopyBase(
93  TViewDst & viewDst,
94  TViewSrc const & viewSrc,
95  TExtent const & extent) :
96  m_extent(extent::getExtentVec(extent)),
97  m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem))),
98 #if (!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
99  m_dstExtent(extent::getExtentVec(viewDst)),
100  m_srcExtent(extent::getExtentVec(viewSrc)),
101 #endif
102  m_dstPitchBytes(mem::view::getPitchBytesVec(viewDst)),
103  m_srcPitchBytes(mem::view::getPitchBytesVec(viewSrc)),
104 
105  m_dstMemNative(reinterpret_cast<std::uint8_t *>(mem::view::getPtrNative(viewDst))),
106  m_srcMemNative(reinterpret_cast<std::uint8_t const *>(mem::view::getPtrNative(viewSrc)))
107  {
108  assert((vec::cast<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
109  assert((vec::cast<SrcSize>(m_extent) <= m_srcExtent).foldrAll(std::logical_or<bool>()));
110  assert(m_extentWidthBytes <= m_dstPitchBytes[TDim::value - 1u]);
111  assert(m_extentWidthBytes <= m_srcPitchBytes[TDim::value - 1u]);
112  }
113 
114 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
115  //-----------------------------------------------------------------------------
116  //!
117  //-----------------------------------------------------------------------------
118  ALPAKA_FN_HOST auto printDebug() const
119  -> void
120  {
121  std::cout << BOOST_CURRENT_FUNCTION
122  << " e: " << m_extent
123  << " ewb: " << this->m_extentWidthBytes
124  << " de: " << m_dstExtent

126  << " dpitchb: " << m_dstPitchBytes
127  << " se: " << m_srcExtent
128  << " sptr: " << reinterpret_cast<void const *>(m_srcMemNative)
129  << " spitchb: " << m_srcPitchBytes
130  << std::endl;
131  }
132 #endif

149  //#############################################################################
150  //! The CPU device ND memory copy task.
151  //#############################################################################
152  template<
153  typename TDim,
154  typename TViewDst,
155  typename TViewSrc,
156  typename TExtent>
157  struct TaskCopy : public TaskCopyBase<TDim, TViewDst, TViewSrc, TExtent>
158  {
159  using DimMin1 = dim::DimInt<TDim::value - 1u>;
160 
161  //-----------------------------------------------------------------------------
162  //! Constructor.
163  //-----------------------------------------------------------------------------
164  using TaskCopyBase<TDim, TViewDst, TViewSrc, TExtent>::TaskCopyBase;
165 
166  //-----------------------------------------------------------------------------
167  //!
168  //-----------------------------------------------------------------------------
169  ALPAKA_FN_HOST auto operator()() const
170  -> void
171  {
172  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
173 
174 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
175  this->printDebug();
176 #endif
177  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one iteration.
178  using ExtentSize = typename TaskCopyBase<TDim, TViewDst, TViewSrc, TExtent>::ExtentSize;
179  vec::Vec<DimMin1, ExtentSize> const extentWithoutInnermost(vec::subVecBegin<DimMin1>(this->m_extent));
180  // [z, y, x] -> [y, x] because the z pitch (the full size of the buffer) is not required.
181  using DstSize = typename TaskCopyBase<TDim, TViewDst, TViewSrc, TExtent>::DstSize;
182  vec::Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(vec::subVecEnd<DimMin1>(this->m_dstPitchBytes));
183  using SrcSize = typename TaskCopyBase<TDim, TViewDst, TViewSrc, TExtent>::SrcSize;
184  vec::Vec<DimMin1, SrcSize> const srcPitchBytesWithoutOutmost(vec::subVecEnd<DimMin1>(this->m_srcPitchBytes));
185 
186  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
187  {
188  meta::ndLoopIncIdx(
189  extentWithoutInnermost,
190  [&](vec::Vec<DimMin1, ExtentSize> const & idx)
191  {
192  std::memcpy(
193  reinterpret_cast<void *>(this->m_dstMemNative + (vec::cast<DstSize>(idx) * dstPitchBytesWithoutOutmost).foldrAll(std::plus<DstSize>())),
194  reinterpret_cast<void const *>(this->m_srcMemNative + (vec::cast<SrcSize>(idx) * srcPitchBytesWithoutOutmost).foldrAll(std::plus<SrcSize>())),
195  static_cast<std::size_t>(this->m_extentWidthBytes));
196  });
197  }
198  }
199  };
