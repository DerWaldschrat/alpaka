107 //-----------------------------------------------------------------------------
108 //! All functions that can be used on an accelerator have to be attributed with ALPAKA_FN_ACC_CUDA_ONLY or ALPAKA_FN_ACC.
109 //!
110 //! Usage:
111 //! ALPAKA_FN_ACC
112 //! auto add(std::int32_t a, std::int32_t b)
113 //! -> std::int32_t;
114 //-----------------------------------------------------------------------------
115 #if BOOST_LANG_CUDA
116  #define ALPAKA_FN_ACC_CUDA_ONLY __device__
117  #define ALPAKA_FN_ACC_NO_CUDA __host__
118  #if defined(ALPAKA_ACC_GPU_CUDA_ONLY_MODE)
119  #define ALPAKA_FN_ACC __device__
120  #else
121  #define ALPAKA_FN_ACC __device__ __host__
122  #endif
123  #define ALPAKA_FN_HOST_ACC __device__ __host__
124  #define ALPAKA_FN_HOST __host__
125 #else
126  // NOTE: ALPAKA_FN_ACC_CUDA_ONLY should not be defined to cause build failures when CUDA only functions are used and CUDA is disabled.
127  // However, this also destroys syntax highlighting.
128  #define ALPAKA_FN_ACC_CUDA_ONLY
129  #define ALPAKA_FN_ACC_NO_CUDA
130  #define ALPAKA_FN_ACC
131  #define ALPAKA_FN_HOST_ACC
132  #define ALPAKA_FN_HOST
133 #endif
