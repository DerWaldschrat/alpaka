40  //! build a single class to inherit from different atomic implementations
41  //
42  // This implementation inherit from all three hierarchies.
43  // The multiple usage of the same type for different levels is allowed.
44  // The class provide the feature that each atomic operation can be focused
45  // to a hierarchy level in Alpaka. A operation to a hierarchy is independent
46  // to the memory hierarchy.
47  //
48  // \tparam TGridAtomic atomic implementation for atomic operations between grids within a device
49  // \tparam TBlockAtomic atomic implementation for atomic operations between blocks within a grid
50  // \tparam TThreadAtomic atomic implementation for atomic operations between threads within a block
51  //#############################################################################
52  template<
53  typename TGridAtomic,
54  typename TBlockAtomic,
55  typename TThreadAtomic
56  >
57  class AtomicHierarchy :
58  public TGridAtomic,
59  public std::conditional<
60  std::is_same<TGridAtomic,TBlockAtomic>::value,
61  atomicHierarchy::Empty1,
62  TBlockAtomic
63  >::type,
64  public std::conditional<
65  std::is_same<TGridAtomic,TThreadAtomic>::value ||
66  std::is_same<TBlockAtomic,TThreadAtomic>::value,
67  atomicHierarchy::Empty2,
68  TThreadAtomic
69  >::type
70  {
71  public:
72  using UsedAtomicHierarchies = AtomicHierarchy<
73  TGridAtomic,
74  TBlockAtomic,
75  TThreadAtomic
76  >;
77  };
