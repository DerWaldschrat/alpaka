179  template<
180  typename THierarchy>
181  struct AtomicOp<
182  op::Add,
183  atomic::AtomicCudaBuiltIn,
184  double,
185  THierarchy>
186  {
187  //-----------------------------------------------------------------------------
188  //
189  //-----------------------------------------------------------------------------
190  ALPAKA_FN_ACC_CUDA_ONLY static auto atomicOp(
191  atomic::AtomicCudaBuiltIn const &,
192  double * const addr,
193  double const & value)
194  -> double
195  {
196  // Code from: http://docs.nvidia.com/cuda/cuda-c-programming-guide/#atomic-functions
197 
198  unsigned long long int * address_as_ull(reinterpret_cast<unsigned long long int *>(addr));
199  unsigned long long int old(*address_as_ull);
200  unsigned long long int assumed;
201  do
202  {
203  assumed = old;
204  old = atomicCAS(address_as_ull, assumed, __double_as_longlong(value + __longlong_as_double(assumed)));
205  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
206  }
207  while(assumed != old);
208  return __longlong_as_double(old);
209  }
210  };
