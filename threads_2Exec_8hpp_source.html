122  //! Executes the kernel function object.
123  //-----------------------------------------------------------------------------
124  template<
125  typename TWorkDiv,
126  typename TKernelFnObj,
127  typename... TArgs>
128  ALPAKA_FN_HOST auto operator()(
129  TWorkDiv const & workDiv,
130  TKernelFnObj const & kernelFnObj,
131  TArgs const & ... args) const
132  -> void
133  {
134  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
135 
136  static_assert(
137  dim::Dim<TWorkDiv>::value == TDim::value,
138  "The work division and the executor have to of the same dimensionality!");
139 
140  auto const vuiGridBlockExtents(
141  workdiv::getWorkDiv<Grid, Blocks>(workDiv));
142  auto const vuiBlockThreadExtents(
143  workdiv::getWorkDiv<Block, Threads>(workDiv));
144 
145  auto const uiBlockSharedExternMemSizeBytes(
146  kernel::getBlockSharedExternMemSizeBytes<
147  typename std::decay<TKernelFnObj>::type,
148  AccCpuThreads<TDim, TSize>>(
149  vuiBlockThreadExtents,
150  args...));
151 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
152  std::cout << BOOST_CURRENT_FUNCTION
153  << " BlockSharedExternMemSizeBytes: " << uiBlockSharedExternMemSizeBytes << " B"
154  << std::endl;
155 #endif
156  AccCpuThreads<TDim, TSize> acc(workDiv);
157 
158  if(uiBlockSharedExternMemSizeBytes > 0u)
159  {
160  acc.m_vuiExternalSharedMem.reset(
161  reinterpret_cast<uint8_t *>(
162  boost::alignment::aligned_alloc(16u, uiBlockSharedExternMemSizeBytes)));
163  }
164 
165  auto const uiNumThreadsInBlock(vuiBlockThreadExtents.prod());
166  ThreadPool threadPool(uiNumThreadsInBlock, uiNumThreadsInBlock);
167 
168  // Bind the kernel and its arguments to the grid block function.
169  auto boundGridBlockExecHost(std::bind(
170  &ExecCpuThreadsImpl<TDim, TSize>::gridBlockExecHost<TKernelFnObj, TArgs...>,
171  std::ref(acc),
172  std::placeholders::_1,
173  std::ref(vuiBlockThreadExtents),
174  std::ref(threadPool),
175  std::ref(kernelFnObj),
176  std::ref(args)...));
177 
178  // Execute the blocks serially.
179  ndLoop(
180  vuiGridBlockExtents,
181  boundGridBlockExecHost);
182 
183  // After all blocks have been processed, the external shared memory has to be deleted.
184  acc.m_vuiExternalSharedMem.reset();
185  }
186 
187  private:
188  //-----------------------------------------------------------------------------
189  //! The function executed for each grid block.
190  //-----------------------------------------------------------------------------
191  template<
192  typename TKernelFnObj,
193  typename... TArgs>
194  ALPAKA_FN_HOST static auto gridBlockExecHost(
195  AccCpuThreads<TDim, TSize> & acc,
196  Vec<TDim, TSize> const & vuiGridBlockIdx,
197  Vec<TDim, TSize> const & vuiBlockThreadExtents,
198  ThreadPool & threadPool,
199  TKernelFnObj const & kernelFnObj,
200  TArgs const & ... args)
201  -> void
202  {
203  // The futures of the threads in the current block.
204  std::vector<std::future<void>> vFuturesInBlock;
205 
206  // Set the index of the current block
207  acc.m_vuiGridBlockIdx = vuiGridBlockIdx;
208 
209  // Bind the kernel and its arguments to the host block thread execution function.
210  auto boundBlockThreadExecHost(std::bind(
211  &ExecCpuThreadsImpl<TDim, TSize>::blockThreadExecHost<TKernelFnObj, TArgs...>,
212  std::ref(acc),
213  std::ref(vFuturesInBlock),
214  std::placeholders::_1,
215  std::ref(threadPool),
216  std::ref(kernelFnObj),
217  std::ref(args)...));
218  // Execute the block threads in parallel.
219  ndLoop(
220  vuiBlockThreadExtents,
221  boundBlockThreadExecHost);
222 
223  // Wait for the completion of the block thread kernels.
224  std::for_each(
225  vFuturesInBlock.begin(),
226  vFuturesInBlock.end(),
227  [](std::future<void> & t)
228  {
229  t.wait();
230  }
231  );
232  // Clean up.
233  vFuturesInBlock.clear();
234 
235  acc.m_mThreadsToIndices.clear();
236  acc.m_mThreadsToBarrier.clear();
237 
238  // After a block has been processed, the shared memory has to be deleted.
239  block::shared::freeMem(acc);
240  }
241  //-----------------------------------------------------------------------------
242  //! The function executed for each block thread on the host.
243  //-----------------------------------------------------------------------------
244  template<
245  typename TKernelFnObj,
246  typename... TArgs>
247  ALPAKA_FN_HOST static auto blockThreadExecHost(
248  AccCpuThreads<TDim, TSize> & acc,
249  std::vector<std::future<void>> & vFuturesInBlock,
250  Vec<TDim, TSize> const & vuiBlockThreadIdx,
251  ThreadPool & threadPool,
252  TKernelFnObj const & kernelFnObj,
253  TArgs const & ... args)
254  -> void
255  {
256  // Bind the arguments to the accelerator block thread execution function.
257  // The vuiBlockThreadIdx is required to be copied in because the variable will get changed for the next iteration/thread.
258  auto boundBlockThreadExecAcc(
259  [&, vuiBlockThreadIdx]()
260  {
261  blockThreadExecAcc(
262  acc,
263  vuiBlockThreadIdx,
264  kernelFnObj,
265  args...);
266  });
267  // Add the bound function to the block thread pool.
268  vFuturesInBlock.emplace_back(
269  threadPool.enqueueTask(
270  boundBlockThreadExecAcc));
271  }
272  //-----------------------------------------------------------------------------
273  //! The thread entry point on the accelerator.
274  //-----------------------------------------------------------------------------
275  template<
276  typename TKernelFnObj,
277  typename... TArgs>
278  ALPAKA_FN_HOST static auto blockThreadExecAcc(
279  AccCpuThreads<TDim, TSize> & acc,
280  Vec<TDim, TSize> const & vuiBlockThreadIdx,
281  TKernelFnObj const & kernelFnObj,
282  TArgs const & ... args)
283  -> void
284  {
285  // We have to store the thread data before the kernel is calling any of the methods of this class depending on them.
286  auto const idThread(std::this_thread::get_id());
287 
288  // Set the master thread id.
289  if(vuiBlockThreadIdx.sum() == 0)
290  {
291  acc.m_idMasterThread = idThread;
292  }
293 
294  // We can not use the default syncBlockThreads here because it searches inside m_mThreadsToBarrier for the thread id.
295  // Concurrently searching while others use emplace is unsafe!
296  typename std::map<std::thread::id, TSize>::iterator itThreadToBarrier;
297 
298  {
299  // The insertion of elements has to be done one thread at a time.
300  std::lock_guard<std::mutex> lock(acc.m_mtxMapInsert);
301 
302  // Save the thread id, and index.
303  acc.m_mThreadsToIndices.emplace(idThread, vuiBlockThreadIdx);
304  itThreadToBarrier = acc.m_mThreadsToBarrier.emplace(idThread, 0).first;
305  }
306 
307  // Sync all threads so that the maps with thread id's are complete and not changed after here.
308  acc.syncBlockThreads(itThreadToBarrier);
309 
310  // Execute the kernel itself.
311  kernelFnObj(
312  const_cast<AccCpuThreads<TDim, TSize> const &>(acc),
313  args...);
314 
315  // We have to sync all threads here because if a thread would finish before all threads have been started,
316  // a new thread could get the recycled (then duplicate) thread id!
317  acc.syncBlockThreads(itThreadToBarrier);
318  }
319  };
320  }
