175  //! Constructor.
176  //-----------------------------------------------------------------------------
177  ALPAKA_FN_ACC_CUDA_ONLY UniformReal() = default;
178 
179  //-----------------------------------------------------------------------------
180  //! Call operator.
181  //-----------------------------------------------------------------------------
182  template<
183  typename TGenerator>
184  ALPAKA_FN_ACC_CUDA_ONLY auto operator()(
185  TGenerator & generator)
186  -> float
187  {
188  // (0.f, 1.0f]
189  float const fUniformRand(curand_uniform(&generator.m_State));
190  // NOTE: (1.0f - curand_uniform) does not work, because curand_uniform seems to return denormalized floats around 0.f.
191  // [0.f, 1.0f)
192  return fUniformRand * static_cast<float>( fUniformRand != 1.0f );
193  }
194  };
