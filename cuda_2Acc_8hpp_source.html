147  //! \return The pointer to the externally allocated block shared memory.

149  template<
150  typename T>
151  ALPAKA_FN_ACC_CUDA_ONLY auto getBlockSharedExternMem() const
152  -> T *
153  {
154  // Because unaligned access to variables is not allowed in device code,
155  // we have to use the widest possible type to have all types aligned correctly.
156  // See: http://docs.nvidia.com/cuda/cuda-c-programming-guide/index.html#shared
157  // http://docs.nvidia.com/cuda/cuda-c-programming-guide/index.html#vector-types

159  return reinterpret_cast<T *>(shMem);
