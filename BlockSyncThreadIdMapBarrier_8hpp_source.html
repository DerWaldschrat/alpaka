83  //-----------------------------------------------------------------------------
84  //! Syncs all threads in the current block.
85  //-----------------------------------------------------------------------------
86  ALPAKA_FN_ACC_NO_CUDA auto syncBlockThreads(
87  ThreadIdToBarrierIdxMapIterator const & itFind) const
88  -> void
89  {
90  assert(itFind != m_threadIdToBarrierIdxMap.end());
91 
92  auto & barrierIdx(itFind->second);
93  TSize const modBarrierIdx(barrierIdx % 2);
94 
95  auto & bar(m_barriers[modBarrierIdx]);
96 
97  // (Re)initialize a barrier if this is the first thread to reach it.
98  // DCLP: Double checked locking pattern for better performance.
99  if(bar.getThreadCount() == 0)
100  {
101  std::lock_guard<std::mutex> lock(m_mtxBarrier);
102  if(bar.getThreadCount() == 0)
103  {
104  bar.reset(m_blockThreadCount);
105  }
106  }
107 
108  // Wait for the barrier.
109  bar.wait();
110  ++barrierIdx;
111  }
112 
113  TSize const & m_blockThreadCount; //!< The number of threads per block the barrier has to wait for.
114 
115  ThreadIdToBarrierIdxMap & m_threadIdToBarrierIdxMap;
116  //!< We have to keep the current and the last barrier because one of the threads can reach the next barrier before a other thread was wakeup from the last one and has checked if it can run.
117  Barrier mutable m_barriers[2]; //!< The barriers for the synchronization of threads.
118  std::mutex mutable m_mtxBarrier;
