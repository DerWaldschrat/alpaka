60  //! \param uiDividend The dividend.
61  //! \return The biggest number that satisfies the following conditions:
62  //! 1) uiDividend/ret==0

64  //-----------------------------------------------------------------------------
65  template<
66  typename T,
67  typename = typename std::enable_if<std::is_integral<T>::value>::type>
68  ALPAKA_FN_HOST auto nextDivisorLowerOrEqual(
69  T const & uiMaxDivisor,
70  T const & uiDividend)
71  -> T
72  {
73  T uiDivisor(uiMaxDivisor);
74 
75  core::assertValueUnsigned(uiDividend);
76  core::assertValueUnsigned(uiMaxDivisor);
77  assert(uiDividend <= uiMaxDivisor);
78 
79  while((uiDividend%uiDivisor) != 0)
80  {
81  --uiDivisor;
82  }
83 
84  return uiDivisor;
85  }

118  //! Subdivides the given grid thread extents into blocks restricted by:
119  //! 1. The maximum block thread extents
120  //! 2. The maximum block thread count
121  //! 3. The requirement of the block thread extents to divide the grid thread extents without remainder
122  //! 4. The requirement of the block extents.
123  //!
124  //! \param vuiGridThreadExtents
125  //! The full extents of threads in the grid.
126  //! \param vuiMaxBlockThreadExtents
127  //! The maximum extents of threads in a block.
128  //! \param uiMaxBlockThreadsCount
129  //! The maximum number of threads in a block.
130  //! \param bRequireBlockThreadExtentsToDivideGridThreadExtents
131  //! If this is true, the grid thread extents will be multiples of the corresponding block thread extents.
132  //! NOTE: If this is true and vuiGridThreadExtents is prime (or otherwise bad chosen) in a dimension, the block thread extent will be one in this dimension.
133  //! \param eGridBlockExtentsSubDivRestrictions
134  //! The grid block extent subdivision restrictions.
135  //-----------------------------------------------------------------------------
136  template<
137  typename TDim,
138  typename TSize>
139  ALPAKA_FN_HOST auto subDivideGridThreads(
140  Vec<TDim, TSize> const & vuiGridThreadExtents,
141  Vec<TDim, TSize> const & vuiMaxBlockThreadExtents,
142  TSize const & uiMaxBlockThreadsCount,
143  bool bRequireBlockThreadExtentsToDivideGridThreadExtents = true,
144  GridBlockExtentsSubDivRestrictions eGridBlockExtentsSubDivRestrictions = GridBlockExtentsSubDivRestrictions::Unrestricted)
145  -> workdiv::WorkDivMembers<TDim, TSize>
146  {
147  // Assert valid input.
148  assert(uiMaxBlockThreadsCount>0u);
149  for(typename TDim::value_type i(0u); i<TDim::value; ++i)
150  {
151  assert(vuiGridThreadExtents[i]>0u);
152  assert(vuiMaxBlockThreadExtents[i]>0u);
153  }
154 
155  // Initialize the block thread extents with the maximum possible.
156  auto vuiBlockThreadExtents(vuiMaxBlockThreadExtents);
157 
158  // Restrict the max block thread extents with the grid thread extents.
159  // This removes dimensions not required in the given grid thread extents.
160  // This has to be done before the uiMaxBlockThreadsCount clipping to get the maximum correctly.
161  for(typename TDim::value_type i(0); i<TDim::value; ++i)
162  {
163  vuiBlockThreadExtents[i] = std::min(vuiBlockThreadExtents[i], vuiGridThreadExtents[i]);
164  }
165 
166  // For equal block thread extents, restrict it to its minimum component.
167  // For example (512, 256, 1024) will get (256, 256, 256).
168  if(eGridBlockExtentsSubDivRestrictions == GridBlockExtentsSubDivRestrictions::EqualExtents)
169  {
170  auto const uiMinBlockThreadExtent(vuiBlockThreadExtents.min());
171  for(typename TDim::value_type i(0u); i<TDim::value; ++i)
172  {
173  vuiBlockThreadExtents[i] = uiMinBlockThreadExtent;
174  }
175  }
176 
177  // Adjust vuiBlockThreadExtents if its product is too large.
178  if(vuiBlockThreadExtents.prod() > uiMaxBlockThreadsCount)
179  {
180  // Satisfy the following equation:
181  // uiMaxBlockThreadsCount >= vuiBlockThreadExtents.prod()
182  // For example 1024 >= 512 * 512 * 1024
183 
184  // For equal block thread extent this is easily the nth root of uiMaxBlockThreadsCount.
185  if(eGridBlockExtentsSubDivRestrictions == GridBlockExtentsSubDivRestrictions::EqualExtents)
186  {
187  double const fNthRoot(std::pow(uiMaxBlockThreadsCount, 1.0/static_cast<double>(TDim::value)));
188  TSize const uiNthRoot(static_cast<TSize>(fNthRoot));
189  for(typename TDim::value_type i(0u); i<TDim::value; ++i)
190  {
191  vuiBlockThreadExtents[i] = uiNthRoot;
192  }
193  }
194  else if(eGridBlockExtentsSubDivRestrictions == GridBlockExtentsSubDivRestrictions::CloseToEqualExtents)
195  {
196  // Very primitive clipping. Just halve the largest value until it fits.
197  while(vuiBlockThreadExtents.prod() > uiMaxBlockThreadsCount)
198  {
199  auto const uiMaxElemIdx(vuiBlockThreadExtents.maxElem());
200  vuiBlockThreadExtents[uiMaxElemIdx] = vuiBlockThreadExtents[uiMaxElemIdx] / 2u;
201  }
202  }
203  else
204  {
205  // Very primitive clipping. Just halve the smallest value until it fits.
206  while(vuiBlockThreadExtents.prod() > uiMaxBlockThreadsCount)
207  {
208  // Compute the minimum element index but ignore ones.
209  // Ones compare always larger to everything else.
210  auto const uiMinElemIdx(
211  static_cast<TSize>(
212  std::distance(
213  &vuiBlockThreadExtents[0],
214  std::min_element(
215  &vuiBlockThreadExtents[0],
216  &vuiBlockThreadExtents[TDim::value-1u],
217  [](TSize const & a, TSize const & b)
218  {
219  // This first case is redundant.
220  /*if((a == 1u) && (b == 1u))
221  {
222  return false;
223  }
224  else */if(a == 1u)
225  {
226  return false;
227  }
228  else if(b == 1u)
229  {
230  return true;
231  }
232  else
233  {
234  return a < b;
235  }
236  }))));
237  vuiBlockThreadExtents[uiMinElemIdx] = vuiBlockThreadExtents[uiMinElemIdx] / 2u;
238  }
239  }
240  }
241 
242  // Make the block thread extents divide the grid thread extents.
243  if(bRequireBlockThreadExtentsToDivideGridThreadExtents)
244  {
245  if(eGridBlockExtentsSubDivRestrictions == GridBlockExtentsSubDivRestrictions::EqualExtents)
246  {
247  // For equal size block extents we have to compute the gcd of all grid thread extents that is less then the current maximal block thread extent.
248  // For this we compute the divisors of all grid thread extents less then the current maximal block thread extent.
249  std::array<std::set<TSize>, TDim::value> gridThreadExtentsDivisors;
250  for(typename TDim::value_type i(0u); i<TDim::value; ++i)
251  {
252  gridThreadExtentsDivisors[i] =
253  detail::allDivisorsLessOrEqual(
254  vuiGridThreadExtents[i],
255  vuiBlockThreadExtents[i]);
256  }
257  // The maximal common divisor of all block thread extents is the optimal solution.
258  std::set<TSize> intersects[2u];
259  for(typename TDim::value_type i(1u); i<TDim::value; ++i)
260  {
261  intersects[(i-1u)%2u] = gridThreadExtentsDivisors[0];
262  intersects[(i)%2u].clear();
263  set_intersection(
264  intersects[(i-1u)%2u].begin(),
265  intersects[(i-1u)%2u].end(),
266  gridThreadExtentsDivisors[i].begin(),
267  gridThreadExtentsDivisors[i].end(),
268  std::inserter(intersects[i%2], intersects[i%2u].begin()));
269  }
270  TSize const uiMaxCommonDivisor(*(--intersects[(TDim::value-1)%2u].end()));
271  for(typename TDim::value_type i(0u); i<TDim::value; ++i)
272  {
273  vuiBlockThreadExtents[i] = uiMaxCommonDivisor;
274  }
275  }
276  else if(eGridBlockExtentsSubDivRestrictions == GridBlockExtentsSubDivRestrictions::CloseToEqualExtents)
277  {
278  for(typename TDim::value_type i(0); i<TDim::value; ++i)
279  {
280  vuiBlockThreadExtents[i] =
281  detail::nextDivisorLowerOrEqual(
282  vuiBlockThreadExtents[i],
283  vuiGridThreadExtents[i]);
284  }
285  }
286  else
287  {
288  for(typename TDim::value_type i(0); i<TDim::value; ++i)
289  {
290  vuiBlockThreadExtents[i] =
291  detail::nextDivisorLowerOrEqual(
292  vuiBlockThreadExtents[i],
293  vuiGridThreadExtents[i]);
294  }
295  }
296  }
297 
298  // Set the grid block extents (rounded to the next integer not less then the quotient.
299  auto vuiGridBlockExtents(Vec<TDim, TSize>::ones());
300  for(typename TDim::value_type i(0); i<TDim::value; ++i)
301  {
302  vuiGridBlockExtents[i] =
303  static_cast<TSize>(
304  std::ceil(static_cast<double>(vuiGridThreadExtents[i])
305  / static_cast<double>(vuiBlockThreadExtents[i])));
306  }
307 
308  return workdiv::WorkDivMembers<TDim, TSize>(vuiGridBlockExtents, vuiBlockThreadExtents);
309  }

367  auto const uiBlockThreadCountMax(devProps.m_uiBlockThreadsCountMax);
368

370  {
371  return false;

373

375  {
376  if((vuiGridBlockExtents[i] == 0)
377  || (vuiBlockThreadExtents[i] == 0)
378  || (vuiBlockThreadExtentsMax[i] < vuiBlockThreadExtents[i]))
379  {
380  return false;
381  }
382  }
383 
384  return true;
385  }
