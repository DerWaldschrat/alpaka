114  //! Executes the kernel function object.
115  //-----------------------------------------------------------------------------
116  ALPAKA_FN_HOST auto operator()() const
117  -> void
118  {
119  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
120 
121  auto const gridBlockExtent(
122  workdiv::getWorkDiv<Grid, Blocks>(*this));
123  auto const blockThreadExtent(
124  workdiv::getWorkDiv<Block, Threads>(*this));
125  auto const threadElemExtent(
126  workdiv::getWorkDiv<Thread, Elems>(*this));
127 
128  // Get the size of the block shared dynamic memory.
129  auto const blockSharedMemDynSizeBytes(
130  meta::apply(
131  [&](TArgs const & ... args)
132  {
133  return
134  kernel::getBlockSharedMemDynSizeBytes<
135  acc::AccCpuOmp2Blocks<TDim, TSize>>(
136  m_kernelFnObj,
137  blockThreadExtent,
138  threadElemExtent,
139  args...);
140  },
141  m_args));
142 
143 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
144  std::cout << BOOST_CURRENT_FUNCTION
145  << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B" << std::endl;
146 #endif
147  // Bind all arguments except the accelerator.
148  // TODO: With C++14 we could create a perfectly argument forwarding function object within the constructor.
149  auto const boundKernelFnObj(
150  meta::apply(
151  [this](TArgs const & ... args)
152  {
153  return
154  std::bind(
155  std::ref(m_kernelFnObj),
156  std::placeholders::_1,
157  std::ref(args)...);
158  },
159  m_args));
160 
161  // The number of blocks in the grid.
162  TSize const numBlocksInGrid(gridBlockExtent.prod());
163  // There is only ever one thread in a block in the OpenMP 2.0 block accelerator.
164  assert(blockThreadExtent.prod() == 1u);
165 
166  // Force the environment to use the given number of threads.
167  int const ompIsDynamic(::omp_get_dynamic());
168  ::omp_set_dynamic(0);
169 
170  // Execute the blocks in parallel.
171  // NOTE: Setting num_threads(number_of_cores) instead of the default thread number does not improve performance.
172  #pragma omp parallel
173  {
174 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
175  // The first thread does some debug logging.
176  if(::omp_get_thread_num() == 0)
177  {
178  int const numThreads(::omp_get_num_threads());
179  std::cout << BOOST_CURRENT_FUNCTION << " omp_get_num_threads: " << numThreads << std::endl;
180  }
181 #endif
182  acc::AccCpuOmp2Blocks<TDim, TSize> acc(
183  *static_cast<workdiv::WorkDivMembers<TDim, TSize> const *>(this),
184  blockSharedMemDynSizeBytes);
185 
186  // NOTE: schedule(static) does not improve performance.
187 #if _OPENMP < 200805 // For OpenMP < 3.0 you have to declare the loop index (a signed integer) outside of the loop header.
188  std::intmax_t iNumBlocksInGrid(static_cast<std::intmax_t>(numBlocksInGrid));
189  std::intmax_t i;
190  #pragma omp for nowait schedule(guided)
191  for(i = 0; i < iNumBlocksInGrid; ++i)
192 #else
193  #pragma omp for nowait schedule(guided)
194  for(TSize i = 0; i < numBlocksInGrid; ++i)
195 #endif
196  {
197  acc.m_gridBlockIdx =
198  core::mapIdx<TDim::value>(
199 #if _OPENMP < 200805
200  Vec<dim::DimInt<1u>, TSize>(static_cast<TSize>(i)),
201 #else
202  Vec<dim::DimInt<1u>, TSize>(i),
203 #endif
204  gridBlockExtent);
205 
206  boundKernelFnObj(
207  acc);
208 
209  // After a block has been processed, the shared memory has to be deleted.
210  block::shared::st::freeMem(acc);
211  }
212  }
213 
214  // Reset the dynamic thread number setting.
215  ::omp_set_dynamic(ompIsDynamic);
216  }
