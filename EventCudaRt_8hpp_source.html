53  //#############################################################################
54  //! The CUDA RT device event implementation.
55  //#############################################################################
56  class EventCudaImpl final
57  {
58  public:
59  //-----------------------------------------------------------------------------
60  //! Constructor.
61  //-----------------------------------------------------------------------------
62  ALPAKA_FN_HOST EventCudaImpl(
63  dev::DevCudaRt const & dev,
64  bool bBusyWait) :
65  m_dev(dev),
66  m_CudaEvent()
67  {
68  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
69 
70  // Set the current device.
71  ALPAKA_CUDA_RT_CHECK(
72  cudaSetDevice(
73  m_dev.m_iDevice));
74  // Create the event on the current device with the specified flags. Valid flags include:
75  // - cudaEventDefault: Default event creation flag.
76  // - cudaEventBlockingSync : Specifies that event should use blocking synchronization.
77  // A host thread that uses cudaEventSynchronize() to wait on an event created with this flag will block until the event actually completes.
78  // - cudaEventDisableTiming : Specifies that the created event does not need to record timing data.
79  // Events created with this flag specified and the cudaEventBlockingSync flag not specified will provide the best performance when used with cudaStreamWaitEvent() and cudaEventQuery().
80  ALPAKA_CUDA_RT_CHECK(
81  cudaEventCreateWithFlags(
82  &m_CudaEvent,
83  (bBusyWait ? cudaEventDefault : cudaEventBlockingSync) | cudaEventDisableTiming));
84  }
85  //-----------------------------------------------------------------------------
86  //! Copy constructor.
87  //-----------------------------------------------------------------------------
88  ALPAKA_FN_HOST EventCudaImpl(EventCudaImpl const &) = delete;
89  //-----------------------------------------------------------------------------
90  //! Move constructor.
91  //-----------------------------------------------------------------------------
92  ALPAKA_FN_HOST EventCudaImpl(EventCudaImpl &&) = default;
93  //-----------------------------------------------------------------------------
94  //! Copy assignment operator.
95  //-----------------------------------------------------------------------------
96  ALPAKA_FN_HOST auto operator=(EventCudaImpl const &) -> EventCudaImpl & = delete;
97  //-----------------------------------------------------------------------------
98  //! Move assignment operator.
99  //-----------------------------------------------------------------------------
100  ALPAKA_FN_HOST auto operator=(EventCudaImpl &&) -> EventCudaImpl & = default;
101  //-----------------------------------------------------------------------------
102  //! Destructor.
103  //-----------------------------------------------------------------------------
104  ALPAKA_FN_HOST ~EventCudaImpl()
105  {
106  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
107 
108  // Set the current device. \TODO: Is setting the current device before cudaEventDestroy required?
109  ALPAKA_CUDA_RT_CHECK(cudaSetDevice(
110  m_dev.m_iDevice));
111  // In case event has been recorded but has not yet been completed when cudaEventDestroy() is called, the function will return immediately
112  // and the resources associated with event will be released automatically once the device has completed event.
113  // -> No need to synchronize here.
114  ALPAKA_CUDA_RT_CHECK(cudaEventDestroy(
115  m_CudaEvent));
116  }
117 
118  public:
119  dev::DevCudaRt const m_dev; //!< The device this event is bound to.
120  cudaEvent_t m_CudaEvent;
121  };
122  }
