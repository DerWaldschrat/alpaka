145  //-----------------------------------------------------------------------------
146  //! Syncs all threads in the current block.
147  //-----------------------------------------------------------------------------
148  ALPAKA_FN_ACC_NO_CUDA auto syncBlockThreads() const
149  -> void
150  {
151  auto const idThread(std::this_thread::get_id());
152  auto const itFind(m_mThreadsToBarrier.find(idThread));
153 
154  syncBlockThreads(itFind);
155  }
156  private:
157  //-----------------------------------------------------------------------------
158  //! Syncs all threads in the current block.
159  //-----------------------------------------------------------------------------
160  ALPAKA_FN_ACC_NO_CUDA auto syncBlockThreads(
161  typename std::map<std::thread::id, TSize>::iterator const & itFind) const
162  -> void
163  {
164  assert(itFind != m_mThreadsToBarrier.end());
165 
166  auto & uiBarrierIdx(itFind->second);
167  TSize const uiModBarrierIdx(uiBarrierIdx % 2);
168 
169  auto & bar(m_abarSyncThreads[uiModBarrierIdx]);
170 
171  // (Re)initialize a barrier if this is the first thread to reach it.
172  // DCLP: Double checked locking pattern for better performance.
173  if(bar.getNumThreadsToWaitFor() == 0)
174  {
175  std::lock_guard<std::mutex> lock(m_mtxBarrier);
176  if(bar.getNumThreadsToWaitFor() == 0)
177  {
178  bar.reset(m_uiNumThreadsPerBlock);
179  }
180  }
181 
182  // Wait for the barrier.
183  bar.wait();
184  ++uiBarrierIdx;
185  }
