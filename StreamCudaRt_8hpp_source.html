46  //! The CUDA RT stream implementation.
47  //#############################################################################
48  class StreamCudaImpl final
49  {
50  public:
51  //-----------------------------------------------------------------------------
52  //! Constructor.
53  //-----------------------------------------------------------------------------
54  StreamCudaImpl(
55  dev::DevCudaRt const & dev) :
56  m_Dev(dev),
57  m_CudaStream()
58  {
59  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
60 
61  // Set the current device.
62  ALPAKA_CUDA_RT_CHECK(cudaSetDevice(
63  m_Dev.m_iDevice));
64  // - cudaStreamDefault: Default stream creation flag.
65  // - cudaStreamNonBlocking: Specifies that work running in the created stream may run concurrently with work in stream 0 (the NULL stream),
66  // and that the created stream should perform no implicit synchronization with stream 0.
67  // Create the stream on the current device.
68  // NOTE: cudaStreamNonBlocking is required to match the semantic implemented in the CPU stream.
69  // It would be too much work to implement implicit default stream synchronization on CPU.
70  ALPAKA_CUDA_RT_CHECK(cudaStreamCreateWithFlags(
71  &m_CudaStream,
72  cudaStreamNonBlocking));
73  }
74  //-----------------------------------------------------------------------------
75  //! Copy constructor.
76  //-----------------------------------------------------------------------------
77  ALPAKA_FN_HOST StreamCudaImpl(StreamCudaImpl const &) = delete;
78  //-----------------------------------------------------------------------------
79  //! Move constructor.
80  //-----------------------------------------------------------------------------
81  ALPAKA_FN_HOST StreamCudaImpl(StreamCudaImpl &&) = default;
82  //-----------------------------------------------------------------------------
83  //! Copy assignment operator.
84  //-----------------------------------------------------------------------------
85  ALPAKA_FN_HOST auto operator=(StreamCudaImpl const &) -> StreamCudaImpl & = delete;
86  //-----------------------------------------------------------------------------
87  //! Move assignment operator.
88  //-----------------------------------------------------------------------------
89  ALPAKA_FN_HOST auto operator=(StreamCudaImpl &&) -> StreamCudaImpl & = default;
90  //-----------------------------------------------------------------------------
91  //! Destructor.
92  //-----------------------------------------------------------------------------
93  ~StreamCudaImpl()
94  {
95  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
96 
97  // Set the current device. \TODO: Is setting the current device before cudaStreamDestroy required?
98  ALPAKA_CUDA_RT_CHECK(cudaSetDevice(
99  m_Dev.m_iDevice));
100  // In case the device is still doing work in the stream when cudaStreamDestroy() is called, the function will return immediately
101  // and the resources associated with stream will be released automatically once the device has completed all work in stream.
102  // -> No need to synchronize here.
103  ALPAKA_CUDA_RT_CHECK(cudaStreamDestroy(
104  m_CudaStream));
105  }
106 
107  public:
108  dev::DevCudaRt const m_Dev; //!< The device this stream is bound to.
109  cudaStream_t m_CudaStream;
110  };

216  // Query is allowed even for streams on non current device.

220  if(ret == cudaSuccess)
221  {
222  return true;
223  }
224  else if(ret == cudaErrorNotReady)
225  {
226  return false;
227  }
228  else
229  {

231  }
