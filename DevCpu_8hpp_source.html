105  //-----------------------------------------------------------------------------
106  //! \return The list of all streams on this device.
107  //-----------------------------------------------------------------------------
108  ALPAKA_FN_HOST auto GetAllAsyncStreamImpls() const noexcept(false)
109  -> std::vector<std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl>>
110  {
111  std::vector<std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl>> vspStreams;
112 
113  std::lock_guard<std::mutex> lk(m_Mutex);
114 
115  for(auto const & pairStream : m_mapStreams)
116  {
117  auto spStream(pairStream.second.lock());
118  if(spStream)
119  {
120  vspStreams.emplace_back(std::move(spStream));
121  }
122  else
123  {
124  throw std::logic_error("One of the streams registered on the device is invalid!");
125  }
126  }
127  return vspStreams;
128  }
129 
130  private:
131  //-----------------------------------------------------------------------------
132  //! Registers the given stream on this device.
133  //! NOTE: Every stream has to be registered for correct functionality of device wait operations!
134  //-----------------------------------------------------------------------------
135  ALPAKA_FN_HOST auto RegisterAsyncStream(std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl> spStreamImpl)
136  -> void
137  {
138  std::lock_guard<std::mutex> lk(m_Mutex);
139 
140  // Register this stream on the device.
141  // NOTE: We have to store the plain pointer next to the weak pointer.
142  // This is necessary to find the entry on unregistering because the weak pointer will already be invalid at that point.
143  m_mapStreams.emplace(spStreamImpl.get(), spStreamImpl);
144  }
145  //-----------------------------------------------------------------------------
146  //! Unregisters the given stream from this device.
147  //-----------------------------------------------------------------------------
148  ALPAKA_FN_HOST auto UnregisterAsyncStream(stream::cpu::detail::StreamCpuAsyncImpl const * const pStream) noexcept(false)
149  -> void
150  {
151  std::lock_guard<std::mutex> lk(m_Mutex);
152 
153  // Unregister this stream from the device.
154  auto const itFind(std::find_if(
155  m_mapStreams.begin(),
156  m_mapStreams.end(),
157  [pStream](std::pair<stream::cpu::detail::StreamCpuAsyncImpl *, std::weak_ptr<stream::cpu::detail::StreamCpuAsyncImpl>> const & pair)
158  {
159  return (pStream == pair.first);
160  }));
161  if(itFind != m_mapStreams.end())
162  {
163  m_mapStreams.erase(itFind);
164  }
165  else
166  {
167  throw std::logic_error("The stream to unregister from the device could not be found in the list of registered streams!");
168  }
169  }
170 
171  private:
172  std::mutex mutable m_Mutex;
173  std::map<stream::cpu::detail::StreamCpuAsyncImpl *, std::weak_ptr<stream::cpu::detail::StreamCpuAsyncImpl>> m_mapStreams;
174  };
175  }
