98  //-----------------------------------------------------------------------------
99  //! \return The list of all streams on this device.
100  //-----------------------------------------------------------------------------
101  ALPAKA_FN_HOST auto GetAllAsyncStreamImpls() const noexcept(false)
102  -> std::vector<std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl>>
103  {
104  std::vector<std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl>> vspStreams;
105 
106  std::lock_guard<std::mutex> lk(m_Mutex);
107 
108  for(auto const & pairStream : m_mapStreams)
109  {
110  auto spStream(pairStream.second.lock());
111  if(spStream)
112  {
113  vspStreams.emplace_back(std::move(spStream));
114  }
115  else
116  {
117  throw std::logic_error("One of the streams registered on the device is invalid!");
118  }
119  }
120  return vspStreams;
121  }
122 
123  private:
124  //-----------------------------------------------------------------------------
125  //! Registers the given stream on this device.
126  //! NOTE: Every stream has to be registered for correct functionality of device wait operations!
127  //-----------------------------------------------------------------------------
128  ALPAKA_FN_HOST auto RegisterAsyncStream(std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl> spStreamImpl)
129  -> void
130  {
131  std::lock_guard<std::mutex> lk(m_Mutex);
132 
133  // Register this stream on the device.
134  // NOTE: We have to store the plain pointer next to the weak pointer.
135  // This is necessary to find the entry on unregistering because the weak pointer will already be invalid at that point.
136  m_mapStreams.emplace(spStreamImpl.get(), spStreamImpl);
137  }
138  //-----------------------------------------------------------------------------
139  //! Unregisters the given stream from this device.
140  //-----------------------------------------------------------------------------
141  ALPAKA_FN_HOST auto UnregisterAsyncStream(stream::cpu::detail::StreamCpuAsyncImpl const * const pStream) noexcept(false)
142  -> void
143  {
144  std::lock_guard<std::mutex> lk(m_Mutex);
145 
146  // Unregister this stream from the device.
147  auto const itFind(std::find_if(
148  m_mapStreams.begin(),
149  m_mapStreams.end(),
150  [pStream](std::pair<stream::cpu::detail::StreamCpuAsyncImpl *, std::weak_ptr<stream::cpu::detail::StreamCpuAsyncImpl>> const & pair)
151  {
152  return (pStream == pair.first);
153  }));
154  if(itFind != m_mapStreams.end())
155  {
156  m_mapStreams.erase(itFind);
157  }
158  else
159  {
160  throw std::logic_error("The stream to unregister from the device could not be found in the list of registered streams!");
161  }
162  }
163 
164  private:
165  std::mutex mutable m_Mutex;
166  std::map<stream::cpu::detail::StreamCpuAsyncImpl *, std::weak_ptr<stream::cpu::detail::StreamCpuAsyncImpl>> m_mapStreams;
167  };

483  //#############################################################################
484  //! The CPU device thread wait specialization.
485  //!
486  //! Blocks until the device has completed all preceding requested tasks.
487  //! Tasks that are enqueued or streams that are created after this call is made are not waited for.
488  //#############################################################################
489  template<>
490  struct CurrentThreadWaitFor<
491  dev::DevCpu>
492  {
493  //-----------------------------------------------------------------------------
494  //
495  //-----------------------------------------------------------------------------
496  ALPAKA_FN_HOST static auto currentThreadWaitFor(
497  dev::DevCpu const & dev)
498  -> void
499  {
500  ALPAKA_DEBUG_FULL_LOG_SCOPE;
501 
502  // Get all the streams on the device at the time of invocation.
503  // All streams added afterwards are ignored.
504  auto vspStreams(
505  dev.m_spDevCpuImpl->GetAllAsyncStreamImpls());
506 
507  // Enqueue an event in every asynchronous stream on the device.
508  // \TODO: This should be done atomically for all streams.
509  // Furthermore there should not even be a chance to enqueue something between getting the streams and adding our wait events!
510  std::vector<event::EventCpu> vEvents;
511  for(auto && spStream : vspStreams)
512  {
513  vEvents.emplace_back(dev);
514  stream::enqueue(spStream, vEvents.back());
515  }
516 
517  // Now wait for all the events.
518  for(auto && event : vEvents)
519  {
520  wait::wait(event);
521  }
522  }
523  };
524  }
