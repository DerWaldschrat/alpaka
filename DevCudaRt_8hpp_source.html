131  std::stringstream ssErr;

133  throw std::runtime_error(ssErr.str());
134  }
135 
136  // Initialize the cuda runtime.
137  //init();
138 
139  // Try all devices if the given one is unusable.
140  for(std::size_t iDeviceOffset(0); iDeviceOffset < uiNumDevices; ++iDeviceOffset)
141  {
142  std::size_t const iDevice((uiIdx + iDeviceOffset) % uiNumDevices);
143 
144  if(isDevUsable(iDevice))

146  dev.m_iDevice = static_cast<int>(iDevice);
147 
148  // Log this device.
149 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL

151  ALPAKA_CUDA_RT_CHECK(cudaGetDeviceProperties(&devProp, dev.m_iDevice));
152 #endif

154  printDeviceProperties(devProp);
155 #elif ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
156  std::cout << BOOST_CURRENT_FUNCTION << devProp.name << std::endl;
157 #endif
158  return dev;
159  }
160  }
161 
162  // If we came until here, none of the devices was usable.
163  std::stringstream ssErr;
164  ssErr << "Unable to return device handle for device " << uiIdx << " because none of the " << uiNumDevices << " CUDA devices is usable!";
