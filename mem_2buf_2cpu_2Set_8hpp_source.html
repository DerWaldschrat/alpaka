122  //#############################################################################
123  //! The CPU device ND memory set task.
124  //#############################################################################
125  template<
126  typename TDim,
127  typename TView,
128  typename TExtent>
129  struct TaskSet : public TaskSetBase<TDim, TView, TExtent>
130  {
131  using DimMin1 = dim::DimInt<TDim::value - 1u>;
132 
133  //-----------------------------------------------------------------------------
134  //! Constructor.
135  //-----------------------------------------------------------------------------
136  using TaskSetBase<TDim, TView, TExtent>::TaskSetBase;
137 
138  //-----------------------------------------------------------------------------
139  //!
140  //-----------------------------------------------------------------------------
141  ALPAKA_FN_HOST auto operator()() const
142  -> void
143  {
144  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
145 
146 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
147  this->printDebug();
148 #endif
149  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one iteration.
150  using ExtentSize = typename TaskSetBase<TDim, TView, TExtent>::ExtentSize;
151  vec::Vec<DimMin1, ExtentSize> const extentWithoutInnermost(vec::subVecBegin<DimMin1>(this->m_extent));
152  // [z, y, x] -> [y, x] because the z pitch (the full size of the buffer) is not required.
153  using DstSize = typename TaskSetBase<TDim, TView, TExtent>::DstSize;
154  vec::Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(vec::subVecEnd<DimMin1>(this->m_dstPitchBytes));
155 
156  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
157  {
158  meta::ndLoopIncIdx(
159  extentWithoutInnermost,
160  [&](vec::Vec<DimMin1, ExtentSize> const & idx)
161  {
162  std::memset(
163  reinterpret_cast<void *>(this->m_dstMemNative + (vec::cast<DstSize>(idx) * dstPitchBytesWithoutOutmost).foldrAll(std::plus<DstSize>())),
164  this->m_byte,
165  static_cast<std::size_t>(this->m_extentWidthBytes));
166  });
167  }
168  }
169  };
