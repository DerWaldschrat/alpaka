92  //-----------------------------------------------------------------------------
93  //! Executes the kernel function object.
94  //-----------------------------------------------------------------------------
95  template<
96  typename TWorkDiv,
97  typename TKernelFnObj,
98  typename... TArgs>
99  ALPAKA_FN_HOST auto operator()(
100  TWorkDiv const & workDiv,
101  TKernelFnObj const & kernelFnObj,
102  TArgs const & ... args) const
103  -> void
104  {
105  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
106 
107  static_assert(
108  dim::Dim<TWorkDiv>::value == TDim::value,
109  "The work division and the executor have to of the same dimensionality!");
110 
111  auto const vuiGridBlockExtents(
112  workdiv::getWorkDiv<Grid, Blocks>(workDiv));
113  auto const vuiBlockThreadExtents(
114  workdiv::getWorkDiv<Block, Threads>(workDiv));
115 
116  auto const uiBlockSharedExternMemSizeBytes(
117  kernel::getBlockSharedExternMemSizeBytes<
118  typename std::decay<TKernelFnObj>::type,
119  acc::AccCpuSerial<TDim, TSize>>(
120  vuiBlockThreadExtents,
121  args...));
122 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
123  std::cout << BOOST_CURRENT_FUNCTION
124  << " BlockSharedExternMemSizeBytes: " << uiBlockSharedExternMemSizeBytes << " B"
125  << std::endl;
126 #endif
127  acc::AccCpuSerial<TDim, TSize> acc(workDiv);
128 
129  if(uiBlockSharedExternMemSizeBytes > 0u)
130  {
131  acc.m_vuiExternalSharedMem.reset(
132  reinterpret_cast<uint8_t *>(
133  boost::alignment::aligned_alloc(16u, uiBlockSharedExternMemSizeBytes)));
134  }
135 
136  // There is only ever one thread in a block in the serial accelerator.
137  assert(vuiBlockThreadExtents.prod() == 1u);
138 
139  // Execute the blocks serially.
140  core::ndLoop(
141  vuiGridBlockExtents,
142  [&](Vec<TDim, TSize> const & vuiBlockThreadIdx)
143  {
144  acc.m_vuiGridBlockIdx = vuiBlockThreadIdx;
145 
146  kernelFnObj(
147  const_cast<acc::AccCpuSerial<TDim, TSize> const &>(acc),
148  args...);
149 
150  // After a block has been processed, the shared memory has to be deleted.
151  block::shared::freeMem(acc);
152  });
153 
154  // After all blocks have been processed, the external shared memory has to be deleted.
155  acc.m_vuiExternalSharedMem.reset();
156  }
157  };
158  }
159  }
