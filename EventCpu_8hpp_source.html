244  //! The CPU async device stream enqueue trait specialization.
245  //#############################################################################
246  template<>
247  struct Enqueue<
248  std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl>,
249  event::EventCpu>
250  {
251  //-----------------------------------------------------------------------------
252  //
253  //-----------------------------------------------------------------------------
254  ALPAKA_FN_HOST static auto enqueue(
255 #if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_CUDA_DEVICE)
256  std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl> & spStreamImpl,
257 #else
258  std::shared_ptr<stream::cpu::detail::StreamCpuAsyncImpl> &,
259 #endif
260  event::EventCpu & event)
261  -> void
262  {
263  // Copy the shared pointer of the event implementation.
264  // This is forwarded to the lambda that is enqueued into the stream to ensure that the event implementation is alive as long as it is enqueued.
265  auto spEventCpuImpl(event.m_spEventCpuImpl);
266 
267  // Setting the event state and enqueuing it has to be atomic.
268  std::lock_guard<std::mutex> lk(spEventCpuImpl->m_Mutex);
269 
270  // This is a invariant: If the event is ready (not enqueued) there can not be anybody waiting for it.
271  assert(!(spEventCpuImpl->m_bIsReady && spEventCpuImpl->m_bIsWaitedFor));
272 
273  // If it is enqueued ...
274  if(!spEventCpuImpl->m_bIsReady)
275  {
276  // ... and somebody is waiting for it, it can NOT be re-enqueued.
277  if(spEventCpuImpl->m_bIsWaitedFor)
278  {
279 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
280  std::cout << BOOST_CURRENT_FUNCTION << "WARNING: The event to enqueue is already enqueued AND waited on. It can NOT be re-enqueued!" << std::endl;
281 #endif
282  return;
283  }
284  // ... and nobody is waiting for it, increment the cancel counter.
285  else
286  {
287  ++spEventCpuImpl->m_canceledEnqueueCount;
288  }
289  }
290  // If it is not enqueued, set its state to enqueued.
291  else
292  {
293  spEventCpuImpl->m_bIsReady = false;
294  }
295 
296  // We can not unlock the mutex here, because the order of events enqueued has to be identical to the call order.
297  // Unlocking here would allow a later enqueue call to complete before this event is enqueued.
298 // Workaround: Clang can not support this when natively compiling device code. See ConcurrentExecPool.hpp.
299 #if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_CUDA_DEVICE)
300  // Enqueue a task that only resets the events flag if it is completed.
301  spStreamImpl->m_workerThread.enqueueTask(
302  [spEventCpuImpl]()
303  {
304  {
305  std::lock_guard<std::mutex> lk2(spEventCpuImpl->m_Mutex);
306  // Nothing to do if it has been re-enqueued to a later position in the queue.
307  if(spEventCpuImpl->m_canceledEnqueueCount > 0)
308  {
309  --spEventCpuImpl->m_canceledEnqueueCount;
310  return;
311  }
312  else
313  {
314  spEventCpuImpl->m_bIsWaitedFor = false;
315  spEventCpuImpl->m_bIsReady = true;
316  }
317  }
318  spEventCpuImpl->m_ConditionVariable.notify_all();
319  });
320 #endif
321  }
322  };
